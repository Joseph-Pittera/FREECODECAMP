function rot13(str) {  
  let result = str
    .split("")
    .reduce((acc, val) => {
      return /[A-Z]/.test(val) 
        ? acc.concat(rot13letter(val))
        : acc.concat(val);
    }, []);
  return result.join("");
}

function rot13letter(letter) {
  return String.fromCharCode((letter.charCodeAt() - 65 + 13) % 26 + 65);
}
